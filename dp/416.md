## 416. 分割等和子集
给定一个只包含正整数的非空数组。是否可以将这个数组分割成两个子集，使得两个子集的元素和相等。

### python使用set去重
```python
'''
思路：看nums中是否有子集和满足 sum(nums)/2
对于nums中的每个元素，有选择和不选择两种情况
如果i-1元素时结果有[a,b,c] 则i元素时结果有[a,b,c,a+nums[i],b+nums[i],c+nums[i]]
每次新结果用set去重
从第一个元素迭代到最后一个元素，如果迭代过程中出现元素等于sum(nums)/2就返回true
（这可以看成优化过的背包问题？）

执行用时：552 ms, 在所有 Python3 提交中击败了84.04% 的用户
内存消耗：14.3 MB, 在所有 Python3 提交中击败了50.19% 的用户
'''

class Solution:
    def canPartition(self, nums: List[int]) -> bool:
        # 不能被2整除就不存在
        if sum(nums) % 2 != 0:
            return False

        target = sum(nums) / 2
        s = set()
        s = {0, nums[0]}
        if target in s:
            return True

        for i in range(1, len(nums)):
            s_new = set()
            for j in s:
                s_new.add(j + nums[i])
                s_new.add(j)
                # 如果新结果中存在目标值
                # 则后续元素都不选择，返回true
                if j + nums[i] == target:
                    return True
            s = s_new

        return False
```

### 上一个方法还原成标准背包问题动态规划解决
```python
'''
因为下一时刻只与上一时刻dp值有关，可以简化为一维数组

执行用时：2108 ms, 在所有 Python3 提交中击败了26.02% 的用户
内存消耗：28.4 MB, 在所有 Python3 提交中击败了5.04% 的用户
'''

class Solution:
    def canPartition(self, nums: List[int]) -> bool:
        if sum(nums) % 2 != 0:
            return False

        target = sum(nums) // 2

        dp = [[0] * (target + 1) for i in range(len(nums))]
        if nums[0] > target:
            return False
        dp[0][0] = 1
        dp[0][nums[0]] = 1

        for i in range(1, len(nums)):
            if nums[i] > target:
                return False
            for j in range(target + 1):
                if j >= nums[i] and dp[i - 1][j - nums[i]]:
                    dp[i][j] = 1
                elif dp[i - 1][j]:
                    dp[i][j] = 1
                
        return dp[len(nums) - 1][target] == 1
```

### c++一维数组动态规划
```c++
// c++用类似python的set方法非常慢，反倒dp很快

//执行用时：152 ms, 在所有 C++ 提交中击败了76.05% 的用户
//内存消耗：12.6 MB, 在所有 C++ 提交中击败了11.75% 的用户

class Solution {
public:
    bool canPartition(vector<int>& nums) {
        int target = 0;
        for(int &i : nums){
            target += i;
        }
        if(target % 2 != 0) return false;
        target = target / 2;
        
        if(target == nums[0]) return true;
        if(target < nums[0]) return false;

        vector<int> dp;
        vector<int> dp_now;
        for(int i = 0; i <= target + 1; i++){
            dp.emplace_back(0);
            dp_now.emplace_back(0);
        }
        dp[0] = 1;
        dp[nums[0]] = 1;

        for(int i = 1; i < nums.size(); i++){
            for(int j = 0; j <= target; j++){
                if(nums[i] > target) return false;
                if(j >= nums[i] && dp[j - nums[i]]){
                    dp_now[j] = 1;
                }else if(dp[j]){
                    dp_now[j] = 1;
                }else{
                    dp_now[j] = 0;
                }
            }
            if(dp_now[target] == 1) return true;
            dp = dp_now;
        }

        return false;
    }
};
```
