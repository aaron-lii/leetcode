# 530. 二叉搜索树的最小绝对差
给你一棵所有节点为非负值的二叉搜索树，请你计算树中【任意】两节点的差的绝对值的最小值。

## python中序遍历，然后计算最小值
```python
# 执行用时：44 ms, 在所有 Python 提交中击败了83.07% 的用户

# Definition for a binary tree node.
# class TreeNode(object):
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution(object):
    def getMinimumDifference(self, root):
        """
        :type root: TreeNode
        :rtype: int
        """
        self.val_list = []

        def dfs(node):
            if not node:
                return
            dfs(node.left)
            self.val_list.append(node.val)
            dfs(node.right)

        dfs(root)

        result = float("inf")
        for i in range(len(self.val_list) - 1):
            result = min(result, self.val_list[i + 1] - self.val_list[i])

        return result
```

## python中序遍历优化
```python
# 执行用时：44 ms, 在所有 Python 提交中击败了83.07% 的用户
# 内存消耗：16.4 MB, 在所有 Python 提交中击败了77.63% 的用户

# Definition for a binary tree node.
# class TreeNode(object):
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution(object):
    def getMinimumDifference(self, root):
        """
        :type root: TreeNode
        :rtype: int
        """
        self.result = float("inf")
        self.pre_val = float("-inf")

        def dfs(node):
            if not node:
                return
            dfs(node.left)
            self.result = min(self.result, node.val - self.pre_val)
            self.pre_val = node.val
            dfs(node.right)

        dfs(root)

        return self.result
```

/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    int result = INT_MAX;
    // 此处注意用INT_MIN会导致下面相减超出int最大值
    int pre_val = -1;

    void dfs(TreeNode* node){
        if(node == nullptr) return;

        dfs(node->left);
        if (pre_val == -1){
            pre_val = node->val;
        }else{
            result = min(result, node->val - pre_val);
            pre_val = node->val;
        }
        dfs(node->right);
    }

    int getMinimumDifference(TreeNode* root) {
        dfs(root);
        return result;
    }
};
